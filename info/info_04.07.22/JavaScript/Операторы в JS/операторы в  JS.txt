

             Инкремент и декремент

let incr = 10,
    decr = 10;

// постпрефиксная форма записи - сначала возыращае сторое значение, а уже потом увеличивает (или уменьшает)

incr++; // оператор инкремента -  увеличивает значение на 1 
decr--; // рператор декремента = уменьшает на 1

console.log(incr); //10

// префиксная форма записи - предпочтительная - сразу возвращае увеличенное значение (или уменьшенное)
++incr;
++decr;

console.log(incr); //11


            Операторы

// && - оператоор "и" - когда каждое из условие верно = true
// || - оператор "или"  - когда выполняется одно из условий = true
//  = - оператор присвоения  (2*4=8 - это присвоиванеи) (2*4 == 8 - true)
// == - оператор равенства (2*4 == 8 - true)  (2*4 = '8' - true) - сравние только по значениею, а не по типу данных
// === - оператор строгого равенства   (2*4 = '8' - false)  - идет строгое сравнение втом числе и по типу данных
// ! - оператор отрицания - обращает значение в обратное   2+2=4 - true; 2+2 != 4 - true
// % - оператор остатка
// + - оператор плюч  2+2=4 - true; 2+2 != 4 - true

логические операторы это 
и-&&
или-|| 

пример

1.
const hamburger = true;
const fries = true;

if (hamburger && fries) {   - условие выполнилось тк true - оба правда - и в консоль выдалост 'OK'
    console.log('OK');
}

2.
const hamburger = true;
const fries = false;

if (hamburger && fries) {   - условие не выполнилось  false тк есть одна неправда  - и в консоль ничего не пошло.
    console.log('OK');
}

3.
const hamburger = true;
const fries = true;

if (hamburger || fries) {   - условие выполнилось тк true - оба правда - и в консоль выдалост 'OK'
    console.log('OK');
}

4.
const hamburger = false;
const fries = true;

if (hamburger || fries) {   - условие выполнилось тк true - есть хотябы одна правда и этого достаточно - и в консоль выдалост 'OK'
    console.log('OK');       
}
