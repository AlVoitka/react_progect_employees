Базовые команды

//1. Alert ( сигнал тревоги, оповещение)

alert('Hi!'); // просто выводит модальнрое окно с заданной информацией



// 2. Confirm (подтверждение, закрепление) 

const result = confirm('Are you here?'); // в result  записывается информация от пользователя
console.log(result);// выводит информацию что выбрал пользователь



//3. Prompt  - когда от ползователя нужно получить ответ или данные

const answer = prompt('Вам есть 18?', ''); // после запятой кавычки обязательно для интерактивности
console.log(answer);// в консоль выведет результат что нажал пользователь . 

//Вся приходящая инфа от пользователя - только в виде строк!!!! 
//А чтобы получить число, нужно поставть + пред prompt = +prompt
